2713 		this.subfieldCodeLength = leader.getSubfieldDelimiterLength();
2714 		var baseAddress = leader.getBaseAddress();
2715 		var dirEntryLength = 3 
2716 			+ leader.getDirectoryLengthLength()
2717 			+ leader.getDirectoryStartLength()
2718 			+ leader.getDirectoryImplLength();
2719 		
2720 		// get record data
2721 		var contentTmp = data.substr(baseAddress);
2722 		
2723 		// MARC wants one-byte characters, so when we have multi-byte UTF-8
2724 		// sequences, add null characters so that the directory shows up right. we
2725 		// can strip the nulls later.
2726 		var content = "";
2727 		for(i=0; i<contentTmp.length; i++) {
2728 			content += contentTmp[i];
2729 			if(contentTmp.charCodeAt(i) > 0x00FFFF) {
2730 				content += "\x00\x00\x00";
2731 			} 
2732 			else if(contentTmp.charCodeAt(i) > 0x0007FF) {
2733 				content += "\x00\x00";
2734 			} 
2735 			else if(contentTmp.charCodeAt(i) > 0x00007F) {
2736 				content += "\x00";
2737 			}
2738 		}
2739 		
2740 		// read directory
2741 		for(var i = 0; i < directory.length; i += dirEntryLength) {
2742 			//Parse directory entry
2743 			var tag = directory.substr(i, 3);
2744 			var len = parseInt(directory.substr(i+3, leader.getDirectoryLengthLength()), 10);
2745 			var pos = parseInt(directory.substr(i+3+leader.getDirectoryLengthLength(),
2746 					leader.getDirectoryStartLength()),
2747 					10);
2748 			//Get field contents
2749 			var data = content.substr(pos, len).replace(/\x00/g, "");
2750 			//Add field
2751 			record.addField(tag).parse(data);
2752 		}
2753 		return record;
2754 	},
2755 	
2756 	/**
2757 	 * Fill a record from preprocessed data.
2758 	 * @param {Array} data an array of objects with the properties "tag" and "value",
2759 	 * holding each a field tag and the corresonding raw value (including the indicator,
2760 	 * if applicable). The record leader's tag must be either empty
2761 	 * or have one of the following conventional values: "000", "LDR".
2762 	 * @returns {Marc.Record} the created record.
2763 	 */
2764 	parseFields : function(data) {
2765 		var record = Marc.Records.newRecord();
2766 		data.forEach(function(field) {
2767 			if(!field.tag || field.tag == "000" || field.tag == "LDR") {
2768 				record.setLeader(field.value);
2769 			}
2770 			else {
2771 				record.addField(field.tag).parse(field.value);
2772 			}
2773 		}, this);
2774 		return record;
2775 	}

2828 /**
2829  * Import binary MARC data
2830  */
2831 function doImport() {
2832 	var maxRecords = Marc.IO.getMaxImportRecords();
2833 	var recordsRead = 0;
2834 	var text;
2835 	var holdOver = "";	// part of the text held over from the last loop
2836 	var importer = Marc.IO.getImporter();
2837 	
2838 	// read in 4096 byte increments
2839 	while(text = Zotero.read(4096)) {	
2840 		var records = text.split(Marc.Delimiters.RECORD_TERMINATOR);
2841 		
2842 		if(records.length > 1) {
2843 			records[0] = holdOver + records[0];
2844 			holdOver = records.pop(); // skip last record, since it's not done
2845 			
2846 			//Record loop
2847 			for(var i in records) {
2848 				//Parse binary data
2849 				var record = importer.parseBinary(records[i], record);
2850 				if(record) {
2851 					//Get converter
2852 					var converter = Marc.Converters.getImportConverter(record);
2853 					
2854 					//Translate
2855 					var item = converter.convert(record);
2856 					item.complete();
2857 					recordsRead++;
2858 					if(maxRecords >= 0 && recordsRead >= maxRecords) {
2859 						break;
2860 					}
2861 				}
2862 			}
2863 		} 
2864 		else {
2865 			holdOver += text;
2866 		}
2867 		if(maxRecords >= 0 && recordsRead >= maxRecords) {
2868 			break;
2869 		}
2870 	}
2871 
2872 }
